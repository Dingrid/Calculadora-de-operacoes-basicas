        private void btsub_Click(object sender, EventArgs e)
        {
            if (string.IsNullOrWhiteSpace(lblvalor1.Text) || string.IsNullOrWhiteSpace(lblvalor1.Text))
                MessageBox.Show("Preencha todas as informações");
            else if (string.IsNullOrWhiteSpace(lblvalor2.Text) || string.IsNullOrWhiteSpace(lblvalor2.Text))
                MessageBox.Show("Preencha todas as informações");
            else
            {
                int val1, val2, resu;
                val1 = Convert.ToInt16(lblvalor1.Text);
                val2 = Convert.ToInt16(lblvalor2.Text);
                resu = val1 - val2;
                lblresu.Text = Convert.ToString(resu);
                lblresu.ForeColor = Color.White;
            }
        }
